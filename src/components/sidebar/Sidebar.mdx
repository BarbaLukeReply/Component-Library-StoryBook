## Usage

```ts
import {Sidebar} from "sld-ui";

const Example = () => {
  const menuItems = [
  {
    name: "Pagina Iniziale",
    link: "/home",
    active: true,
    accordion: false,
    subMenuItems: [],
  },
  {
    name: "Anagrafica",
    link: "",
    active: false,
    accordion: true,
    subMenuItems: [
      {
        name: "Informazioni Generali",
        link: "/general-information",
        active: false,
        accordion: false,
        subMenuItems: [],
      },
      {
        name: "Contatti",
        link: "/contacts",
        active: false,
        accordion: false,
        subMenuItems: [],
      },
      {
        name: "Utente Master/altri utenti",
        link: "/users",
        active: false,
        accordion: false,
        subMenuItems: [],
      },
    ],
  },
  {
    name: "FAQ",
    link: "/faq",
    active: false,
    accordion: false,
    subMenuItems: [],
  },
];
  return (
      <Sidebar
        menuItems={menuItems}
      />
  );
};

export default Example;
```

#### Arguments

- **menuItems** - An array of objects.
