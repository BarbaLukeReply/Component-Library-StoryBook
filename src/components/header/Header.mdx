# Header

Header component with different props.

#### Example

<Canvas of={Header.Sample} />

## Usage

```ts
import {Header} from "sld-ui";
import CompanyLogo from "../../assets/company-logo.jpg";
import LogoutLogo from "../../assets/logout.svg";
import AccountLogo from "../../assets/account.svg";
import ItalyFlag from "../../assets/italy.svg";

const Example = () => {
  return (
      <Header
      userName={"John Doe"}
      isMobile={false}
      menuItems={["Home", "About", "Contact"]}
      companyLogo={"assets/companyLogo.png"}
      logoutLogo={"assets/logoutLogo.png"}
      flags={[{ language: "en", flag: "assets/flags/en.png" }, { language: "fr", flag: "assets/flags/fr.png" }]}
      accountLogo={"assets/accountLogo.png"}
      language={"en"}
      userCompanyName={"User Company Name"}
      companyName={"Company Name"}
      clientCode={"123456"}
    />
  );
};

export default Example;
```

#### Arguments

- **userName**: User name to be displayed in the header.
- **isMobile**: Boolean value to check if the device is mobile or not.
- **menuItems**: Array of menu items to be displayed in the header.
- **companyLogo**: Company logo to be displayed in the header.
- **logoutLogo**: Logout logo to be displayed in the header.
- **flags**: Array of flags to be displayed in the header.
- **accountLogo**: Account logo to be displayed in the header.
- **language**: Language to be displayed in the header.
- **userCompanyName**: User company name to be displayed in the header.
- **companyName**: Company name to be displayed in the header.
- **clientCode**: Client code to be displayed in the header.
